```javascript
// This is a comment: it should be in a lighter cyan color

/**
 * Multi-line comment
 * Class definition with inheritance
 */
class Dragon extends Creature {
  constructor(name, element) {
    super(name);
    this.element = element; // Property: dragon's element
  }

  // Method: describe the dragon
  describe() {
    console.log(`This dragon breathes ${this.element}.`);
  }

  // Static method: belongs to the class, not an instance
  static roar() {
    console.log("The dragon lets out a mighty roar!");
  }
}

// Function: creates a new dragon and describes it
function createDragon(name, element) {
  const dragon = new Dragon(name, element);
  dragon.describe(); // Should call the describe method
  Dragon.roar(); // Static method call
}

// Variable declarations
const fireDragon = "Smaug"; // String: dragon's name
let element = "fire"; // String: element

// Conditionals and operators
if (element === "fire") {
  console.log(`${fireDragon} is a fire dragon.`); // Template string
} else {
  console.log(`${fireDragon} is not a fire dragon.`);
}

// Calling a function
createDragon(fireDragon, element);

// Regex example: match all digits
const pattern = /\d+/g; // Should be in a distinct color

// Keyword: try-catch-finally structure
try {
  throw new Error("Something went wrong");
} catch (error) {
  console.error("Caught an error:", error);
} finally {
  console.log("Always runs, regardless of errors");
}

// Boolean value
const isActive = true; // Boolean: should be styled differently

// Numeric value
let count = 42; // Number: should have its own style
```
